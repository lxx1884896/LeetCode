/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
     vector<vector<int>> res;
    vector<int> tmp;
    vector<vector<int>> pathSum(TreeNode* root, int sum) {
        
      
        
         if(!root) return res;
        
        int sum_leaf=0;
        
        
        
       PostOrderTraverse(root,sum_leaf,sum);
        
       // cout<<flag<<endl;
        
        return res;
        
        
        
    }
    
           
    
 void PostOrderTraverse(TreeNode* T,int sum_leaf,int sum)
{
if (T == NULL)
return  ;
        
        



      if(T )
          
          
         tmp.push_back(T->val);
          sum_leaf+=T->val;
          if(!T->left && !T->right){
              
             // cout<<sum_leaf<<endl;
              
              if(sum_leaf==sum) {
                  
                  res.push_back(tmp);
                  tmp.pop_back();
                  sum_leaf-=T->val;
                  return;
                  
              }
             
             
          }
          
          
          
          PostOrderTraverse(T->left,sum_leaf,sum);
        
          PostOrderTraverse(T->right,sum_leaf,sum);
         tmp.pop_back(); 
         sum_leaf-=T->val;
     
         
          
      
    }
        
   
};
