//自写，需要转化为字符串的
class Solution {
public:
    bool isPalindrome(int x) {
        
        if (x<0) return false;
        if(x==0)  return true;
        string s=to_string(x);
        int len=s.length();
        int i=0;
        for(i=0;i<len/2;){
            
            if(s[i]==s[len-i-1]){
                
                i++;
            }
            
            else break;
        }
        
        if(i==len/2) return true;
        else return false;
        
        
        
        
    }
};
