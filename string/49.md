class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        dict={}
        
        for str_ in strs:
            #tmp.append(x) for x in str:
            str1=list(str_)
            str1.sort()
            tmp="".join(str1)
            
            
            
            #print (type(tmp))
            #print (tmp)
            
            #if tmp not in dict:
                #dict[tmp]=str_
           
            
            dict.setdefault(tmp,[]).append(str_)
        return (list(dict.values()))
